Get started

Install Gulp


									// global scope
									npm i -g gulp-cli
								

									// project folder
									npm i -D gulp

Install Grunt


									// global scope
									npm i -g grunt-cli
								

									// project folder
									npm i -D grunt

Configure a Project

gulpfile.js


								const gulp = require('gulp');

								gulp.task('default', function() {

								 	// task code here

								});
								

									// run it
									gulp

gruntfile.js


								module.exports = function(grunt) {

								 grunt.initConfig({

								  pkg: grunt.file.readJSON('package.json'),
									// task code here

								 })

								}
								

									// run it
									grunt

Example of task

Compile SASS to CSS


							//gulpfile.js
							const gulp = require('gulp');
							const sass = require('gulp-sass');
							//style paths
							const sassFiles = 'assets/styles/sass/**/*.scss';
							const cssDest 	= 'assets/styles/css/';

							gulp.task('styles', function(){

							 gulp.src(sassFiles)
							  .pipe(sass().on('error', sass.logError))
							  .pipe(gulp.dest(cssDest));

							})
							

							//gruntfile.js
							module.exports = function(grunt) {
							 grunt.initConfig({
							  pkg: grunt.file.readJSON('package.json'),
							  sass: {
							   dist: {
							    files: {
							     'style/style.css' : 'sass/style.scss'
							    }
							   }
							  }
							 });
							 grunt.loadNpmTasks('grunt-contrib-sass');
							}
